class Solution {
    public List<Integer> findValidElements(int[] nums) {
        List<Integer> res = new ArrayList<>();
        boolean postMax[] = new boolean[nums.length];
        int max = nums[nums.length-1];
        for(int i=nums.length-1;i>=0;i--){
            if(i==nums.length-1) postMax[i]=true;
            else{
                if(nums[i]>max) {
                    postMax[i]=true;
                    max =nums[i];
                }else{
                    postMax[i]=false;
                }
            }
        }
        max = nums[0];
        for(int i=0;i<nums.length;i++){
            if(i==0 || i==nums.length-1) {
                res.add(nums[i]);
                max = Math.max(max,nums[i]);
            }
            else{
                if(nums[i]>max || postMax[i]){
                    res.add(nums[i]);
                    max = Math.max(max,nums[i]);
                }
            }
        }
        return res;
    }
}
